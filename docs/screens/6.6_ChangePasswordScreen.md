# Экран X.6: Изменение пароля (Change Password Screen)

## Статус реализации

- Статус: Новый план
- Дата создания: Январь 2026

## Обзор

Экран для смены пароля пользователя. Пользователь должен ввести текущий пароль и новый пароль для подтверждения изменения.

## Назначение

Основная цель и функции экрана:

- Изменение пароля пользователя с подтверждением через текущий пароль
- Проверка требований к паролю (минимум символов, сложность)
- Сравнение нового пароля с подтверждением
- Обработка ошибок при изменении пароля
- Уведомление пользователя об успешном изменении

## Компоненты

- Форма ввода текущего пароля
- Форма ввода нового пароля
- Форма ввода подтверждения нового пароля
- Кнопка сохранения изменений
- Загрузочный индикатор

## Навигация

- Откуда переходят на этот экран:
  - Из экрана редактировани япрофиля (`EditProfileScreen`)

- Куда можно перейти с этого экрана:
  - В `EditProfileScreen` (при успешном изменении)
  - Назад (пользователь отменяет действие)

### Use Case

```kotlin
class ChangePasswordUseCase(
    private val userRepository: UserRepository,
) {
    operator fun invoke(
        current: String,
        new: String,
        confirm: String
    ): Result<Unit> = try {
        // Валидация требований к паролю
        if (current.isEmpty()) {
            return Result.failure(ValidationException("Текущий пароль не может быть пустым"))
        }
        if (new.isEmpty()) {
            return Result.failure(ValidationException("Новый пароль не может быть пустым"))
        }
        if (new.length < MIN_PASSWORD_LENGTH) {
            return Result.failure(ValidationException("Пароль должен содержать минимум $MIN_PASSWORD_LENGTH символов"))
        }
        if (new.length > MAX_PASSWORD_LENGTH) {
            return Result.failure(ValidationException("Пароль не должен превышать $MAX_PASSWORD_LENGTH символов"))
        }
        if (!new.equals(confirm)) {
            return Result.failure(ValidationException("Новый пароль должен совпадать с подтверждением"))
        }
        
        // Вызов репозитория
        return userRepository.changePassword(current, new, confirm)
    } catch (e: IOException) {
        Result.failure(NetworkException("Не удалось изменить пароль. Проверьте интернет-соединение", e))
    } catch (e: HttpException) {
        when (e.code()) {
            400 -> Result.failure(ValidationException("Неверные данные формы"))
            401 -> Result.failure(AuthException("Требуется авторизация"))
            404 -> Result.failure(AuthException("Пользователь не найден"))
            500 -> Result.failure(ServerException("Ошибка сервера"))
            else -> Result.failure(ServerException("Неизвестная ошибка"))
        }
    }
    
    companion object {
        private const val MIN_PASSWORD_LENGTH = 6
        private const val MAX_PASSWORD_LENGTH = 32
    }
}
```

### Repository

```kotlin
// UserRepository.kt
class UserRepository(
    private val swApi: SWApi,
    private val userPreferencesRepository: UserPreferencesRepository,
) {
    suspend fun changePassword(
        current: String,
        new: String
    ): Result<ChangePasswordResponse> = try {
        // Проверка авторизации
        if (!userPreferencesRepository.isAuthorized()) {
            return Result.failure(AuthException("Требуется авторизация"))
        }
        
        // Вызов API для изменения пароля
        val response = swApi.changePassword(current, new)
        
        Result.success(response)
    } catch (e: IOException) {
        Result.failure(NetworkException("Не удалось изменить пароль. Проверьте интернет-соединение", e))
    } catch (e: HttpException) {
        // Обработка HTTP исключений
        when (e.code()) {
            400 -> Result.failure(ValidationException("Неверные данные формы"))
            401 -> Result.failure(AuthException("Требуется авторизация"))
            404 -> Result.failure(AuthException("Пользователь не найден"))
            500 -> Result.failure(ServerException("Ошибка сервера"))
            else -> Result.failure(ServerException("Неизвестная ошибка"))
        }
    }
}
```

### ViewModel

```kotlin
// ChangePasswordViewModel.kt
class ChangePasswordViewModel(
    private val changePasswordUseCase: ChangePasswordUseCase,
) : ViewModel() {
    
    // UI State
    private val _uiState = MutableStateFlow<ChangePasswordUiState>(ChangePasswordUiState.Loading)
    val uiState: StateFlow<ChangePasswordUiState> = _uiState.asStateFlow()
    
    // Form fields - public для доступа из UI
    private val _currentPassword = mutableStateOf("")
    val currentPassword: String by _currentPassword
    
    private val _newPassword = mutableStateOf("")
    val newPassword: String by _newPassword
    
    private val _confirmPassword = mutableStateOf("")
    val confirmPassword: String by _confirmPassword
    
    init {
        // Экран готов к работе сразу, без загрузки
        _uiState.value = ChangePasswordUiState.Success(canChangePassword = false)
    }
    
    fun onCurrentPasswordChange(current: String) {
        _currentPassword.value = current
        validateForm()
    }
    
    fun onNewPasswordChange(new: String) {
        _newPassword.value = new
        validateForm()
    }
    
    fun onConfirmPasswordChange(confirm: String) {
        _confirmPassword.value = confirm
        validateForm()
    }
    
    fun onChangePassword() {
        viewModelScope.launch {
            _uiState.value = ChangePasswordUiState.Loading
            
            val result = changePasswordUseCase(
                current = currentPassword,
                new = newPassword
            )
            
            when (result) {
                is Success -> {
                    _uiState.value = ChangePasswordUiState.Success(canChangePassword = true)
                    // Очистить поля
                    _currentPassword.value = ""
                    _newPassword.value = ""
                    _confirmPassword.value = ""
                }
                is Failure -> {
                    _uiState.value = ChangePasswordUiState.Error(result.exceptionOrNull()?.message ?: "Ошибка изменения пароля")
                }
            }
        }
    }
    
    private fun validateForm() {
        // Текущий пароль не может быть пустым
        val isCurrentPasswordValid = currentPassword.isNotEmpty()
        
        // Новый пароль не может быть пустым
        val isNewPasswordValid = newPassword.isNotEmpty()
        
        // Подтверждение должно совпадать с новым паролем
        val isConfirmValid = confirmPassword.isNotEmpty() && 
                          confirmPassword == newPassword
        
        // Проверка длины пароля (минимум 6, максимум 32 символа)
        val isPasswordLengthValid = newPassword.length in 6..32
        
        val canChangePassword = isCurrentPasswordValid && isNewPasswordValid && 
                              isConfirmValid && isPasswordLengthValid
        _uiState.value = if (canChangePassword) {
            ChangePasswordUiState.Success(canChangePassword = true)
        } else {
            ChangePasswordUiState.Success(canChangePassword = false)
        }
    }
}
```
